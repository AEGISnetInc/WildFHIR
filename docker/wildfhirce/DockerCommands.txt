# Useful for setting up a named compose project; shows up in Docker Desktop
docker compose -p wildfhirce-r4-wildfly20-mysql8 create

# Run all Docker containers in a compose definition, project name is defaulted
docker compose up -d

# Build wildfhirce-wildfly image; tag r4-20.0.1.Final-mysql8.0
docker build -t wildfhirce-wildfly:r4-20.0.1.Final-mysql8.0 .

# Build wildfhirce-mysql image; tag r4-mysql8.0
docker build -t wildfhirce-mysql:r4-mysql8.0 .
